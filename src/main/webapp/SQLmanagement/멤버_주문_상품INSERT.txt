-- 2. PD_SIZE 데이터 생성 (모든 상품에 대해 사이즈 생성)
DECLARE
    CURSOR c_products IS 
        SELECT PROD_ID, CATE_CODE FROM PRODUCTS ORDER BY PROD_ID;
        
    -- 사이즈 배열 (220~290mm)
    type num_array is varray(20) of number;
    v_men_sizes   num_array := num_array(250, 255, 260, 265, 270, 275, 280, 285, 290);
    v_women_sizes num_array := num_array(220, 225, 230, 235, 240, 245, 250, 255);
    v_sport_sizes num_array := num_array(220, 225, 230, 235, 240, 245, 250, 255, 260, 265, 270, 275, 280, 285, 290);
    
    v_selected_size NUMBER;
    v_random_idx    NUMBER;
    v_counter       NUMBER := 0;
BEGIN
    FOR r_prod IN c_products LOOP
        -- 모든 상품에 대해 2~3개의 사이즈 옵션 생성 (개수 제한 없음)
        FOR i IN 1..ROUND(DBMS_RANDOM.VALUE(2, 3)) LOOP
            
            -- 카테고리별 사이즈 선택
            IF r_prod.CATE_CODE LIKE 'MEN_%' THEN
                v_random_idx := ROUND(DBMS_RANDOM.VALUE(1, 9)); 
                v_selected_size := v_men_sizes(v_random_idx);
            ELSIF r_prod.CATE_CODE LIKE 'WOMEN_%' THEN
                v_random_idx := ROUND(DBMS_RANDOM.VALUE(1, 8)); 
                v_selected_size := v_women_sizes(v_random_idx);
            ELSE 
                v_random_idx := ROUND(DBMS_RANDOM.VALUE(1, 15)); 
                v_selected_size := v_sport_sizes(v_random_idx);
            END IF;

            -- 데이터 삽입
            INSERT INTO PD_SIZE (OPT_ID, PROD_ID, PD_SIZE)
            VALUES (OPT_SEQ.NEXTVAL, r_prod.PROD_ID, TO_CHAR(v_selected_size));
            
            v_counter := v_counter + 1;
        END LOOP;
    END LOOP;
    
    COMMIT;
    DBMS_OUTPUT.PUT_LINE('PD_SIZE 생성 완료 (전체 상품 대상): ' || v_counter || '건');
END;
/


-- 3. STOCK_LOGS 데이터 생성 (모든 옵션에 대해 재고 생성 + 품절 수량 조절)
DECLARE
    -- 생성된 모든 옵션 조회
    CURSOR c_options IS 
        SELECT OPT_ID, PROD_ID FROM PD_SIZE;
        
    v_change_qty  NUMBER;
    v_change_type VARCHAR2(50);
    v_log_date    DATE;
    v_prod_stock  NUMBER; 
    v_rand_val    NUMBER;
    v_soldout_cnt NUMBER := 0;
BEGIN
    -- 모든 옵션에 대해 최소 1개의 재고 로그를 생성하여 '데이터 없음' 방지
    FOR r_opt IN c_options LOOP
        
        v_rand_val := DBMS_RANDOM.VALUE;
        
        -- **[핵심] 품절 확률 설정 (약 12%)**
        -- 전체 옵션이 약 400개일 때, 12%면 약 48~50개가 품절됨
        IF v_rand_val < 0.12 THEN
            v_prod_stock := 0; -- 품절
            v_change_type := '기타 출고'; -- 전량 소진
            v_change_qty := ROUND(DBMS_RANDOM.VALUE(1, 5));
            v_soldout_cnt := v_soldout_cnt + 1;
        
        -- 재고 임박 (약 10%)
        ELSIF v_rand_val < 0.22 THEN
            v_prod_stock := ROUND(DBMS_RANDOM.VALUE(1, 10));
            v_change_type := '주문 판매';
            v_change_qty := ROUND(DBMS_RANDOM.VALUE(1, 3));
            
        -- 넉넉한 재고 (나머지 약 78%)
        ELSE
            v_prod_stock := ROUND(DBMS_RANDOM.VALUE(11, 100));
            v_change_type := '입고';
            v_change_qty := ROUND(DBMS_RANDOM.VALUE(10, 50));
        END IF;

        -- 날짜: 최근 3개월 이내 (최신 현황처럼 보이게)
        v_log_date := SYSDATE - DBMS_RANDOM.VALUE(0, 90);

        -- 데이터 삽입
        INSERT INTO STOCK_LOGS (
            LOG_ID, PROD_ID, OPT_ID, CHANGE_TYPE, CHANGE_QTY, REG_DATE, PROD_STOCK
        )
        VALUES (
            STOCK_SEQ.NEXTVAL, 
            r_opt.PROD_ID, 
            r_opt.OPT_ID, 
            v_change_type, 
            v_change_qty, 
            v_log_date,
            v_prod_stock
        );
        
        -- (선택) 이력을 풍성하게 하기 위해 30% 확률로 과거 로그 하나 더 추가
        IF DBMS_RANDOM.VALUE < 0.3 THEN
            INSERT INTO STOCK_LOGS (
                LOG_ID, PROD_ID, OPT_ID, CHANGE_TYPE, CHANGE_QTY, REG_DATE, PROD_STOCK
            )
            VALUES (
                STOCK_SEQ.NEXTVAL, 
                r_opt.PROD_ID, 
                r_opt.OPT_ID, 
                '입고', 
                ROUND(DBMS_RANDOM.VALUE(10, 30)), 
                v_log_date - DBMS_RANDOM.VALUE(10, 100), -- 더 과거
                v_prod_stock + ROUND(DBMS_RANDOM.VALUE(1, 10)) -- 과거엔 재고가 달랐음
            );
        END IF;

    END LOOP;

    COMMIT;
    DBMS_OUTPUT.PUT_LINE('STOCK_LOGS 생성 완료');
    DBMS_OUTPUT.PUT_LINE('생성된 품절(0개) 상품 수: 약 ' || v_soldout_cnt || '개');
END;
/
